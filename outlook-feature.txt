```typescript
import { Router } from 'express';
import { pool } from '../db';

const router = Router();

// Get Outlook settings for a project
router.get('/lists/:listId/outlook-settings', async (req, res) => {
  try {
    const { listId } = req.params;
    
    const result = await pool.query(
      'SELECT customer_emails, customer_keywords FROM lists WHERE id = $1',
      [listId]
    );
    
    if (result.rows.length === 0) {
      return res.status(404).json({ error: 'Project not found' });
    }
    
    res.json({
      customer_emails: result.rows[0].customer_emails || [],
      customer_keywords: result.rows[0].customer_keywords || []
    });
  } catch (error) {
    console.error('Error fetching Outlook settings:', error);
    res.status(500).json({ error: 'Failed to fetch Outlook settings' });
  }
});

// Update Outlook settings for a project
router.patch('/lists/:listId/outlook-settings', async (req, res) => {
  try {
    const { listId } = req.params;
    const { customer_emails, customer_keywords } = req.body;
    
    // Validate email format (basic check)
    if (customer_emails && Array.isArray(customer_emails)) {
      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      const invalidEmails = customer_emails.filter(email => !emailRegex.test(email));
      if (invalidEmails.length > 0) {
        return res.status(400).json({ 
          error: `Invalid email addresses: ${invalidEmails.join(', ')}` 
        });
      }
    }
    
    const result = await pool.query(
      `UPDATE lists 
       SET customer_emails = $1, customer_keywords = $2
       WHERE id = $3
       RETURNING id, name, customer_emails, customer_keywords`,
      [customer_emails || [], customer_keywords || [], listId]
    );
    
    if (result.rows.length === 0) {
      return res.status(404).json({ error: 'Project not found' });
    }
    
    res.json(result.rows[0]);
  } catch (error) {
    console.error('Error updating Outlook settings:', error);
    res.status(500).json({ error: 'Failed to update Outlook settings' });
  }
});

export default router;
```

import React, { useState, useEffect } from 'react';
import { getOutlookSettings, type OutlookSettings } from '../api';
import { Mail, ExternalLink, Settings, Calendar } from 'lucide-react';

interface OutlookViewProps {
  selectedListId: string | null;
}

export function OutlookView({ selectedListId }: OutlookViewProps) {
  const [settings, setSettings] = useState<OutlookSettings>({
    customer_emails: [],
    customer_keywords: []
  });
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!selectedListId) {
      setIsLoading(false);
      return;
    }

    setIsLoading(true);
    setError(null);
    
    getOutlookSettings(selectedListId)
      .then(setSettings)
      .catch(err => {
        console.error('Failed to load Outlook settings:', err);
        setError('Failed to load email settings');
      })
      .finally(() => setIsLoading(false));
  }, [selectedListId]);

  if (!selectedListId) {
    return (
      <div className="flex items-center justify-center h-full">
        <div className="text-center text-gray-500">
          <Mail className="w-16 h-16 mx-auto mb-4 opacity-50" />
          <p>Select a project to view email conversations</p>
        </div>
      </div>
    );
  }

  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-full">
        <div className="text-gray-500">Loading email settings...</div>
      </div>
    );
  }

  if (error) {
    return (
      <div className="flex items-center justify-center h-full">
        <div className="text-red-500">{error}</div>
      </div>
    );
  }

  const hasConfiguration = 
    settings.customer_emails.length > 0 || 
    settings.customer_keywords.length > 0;

  const gmailUrl = buildGmailSearchUrl(settings);

  return (
    <div className="flex flex-col items-center justify-center h-full p-8 max-w-2xl mx-auto">
      {/* Header */}
      <div className="text-center mb-8">
        <Mail className="w-20 h-20 mx-auto mb-4 text-blue-600" />
        <h2 className="text-3xl font-semibold mb-2">Email Conversations</h2>
        <p className="text-gray-600">
          View Gmail conversations related to this project
        </p>
      </div>

      {/* Configuration Summary */}
      <div className="w-full bg-gray-50 rounded-lg p-6 mb-6 border border-gray-200">
        <h3 className="font-semibold mb-3 flex items-center">
          <Settings className="w-4 h-4 mr-2" />
          Search Configuration
        </h3>
        
        {hasConfiguration ? (
          <ul className="space-y-2 text-sm text-gray-700">
            {settings.customer_emails.length > 0 && (
              <li>
                <span className="font-medium">Email addresses:</span>
                <ul className="ml-4 mt-1 space-y-1">
                  {settings.customer_emails.map(email => (
                    <li key={email} className="flex items-center">
                      <Mail className="w-3 h-3 mr-2 text-gray-400" />
                      {email}
                    </li>
                  ))}
                </ul>
              </li>
            )}
            
            {settings.customer_keywords.length > 0 && (
              <li>
                <span className="font-medium">Keywords:</span>
                <ul className="ml-4 mt-1 space-y-1">
                  {settings.customer_keywords.map(keyword => (
                    <li key={keyword} className="flex items-center">
                      <span className="text-gray-600">"{keyword}"</span>
                    </li>
                  ))}
                </ul>
              </li>
            )}
            
            <li className="flex items-center pt-2 border-t border-gray-300">
              <Calendar className="w-3 h-3 mr-2 text-gray-400" />
              <span className="font-medium">Date range:</span>
              <span className="ml-2 text-gray-600">Last 40 days</span>
            </li>
          </ul>
        ) : (
          <p className="text-sm text-gray-500 italic">
            No email addresses or keywords configured yet. 
            Configure them in Project Settings to enable email search.
          </p>
        )}
      </div>

      {/* Action Button */}
      <button
        onClick={() => window.open(gmailUrl, '_blank', 'noopener,noreferrer')}
        disabled={!hasConfiguration}
        className={`
          flex items-center gap-2 px-8 py-4 rounded-lg font-medium
          transition-all duration-200 transform hover:scale-105
          ${hasConfiguration
            ? 'bg-blue-600 text-white hover:bg-blue-700 shadow-lg hover:shadow-xl'
            : 'bg-gray-300 text-gray-500 cursor-not-allowed'
          }
        `}
      >
        <Mail className="w-5 h-5" />
        Open in Gmail
        <ExternalLink className="w-4 h-4" />
      </button>

      {/* Help Text */}
      <p className="mt-6 text-sm text-gray-500 text-center max-w-md">
        This will open Gmail in a new tab with emails filtered by the configured 
        addresses and keywords. All filtering happens in Gmail using their search operators.
      </p>
    </div>
  );
}

/**
 * Builds a Gmail search URL from Outlook settings
 */
function buildGmailSearchUrl(settings: OutlookSettings): string {
  const searchParts: string[] = [];
  
  // Add email address filters (from: OR to:)
  if (settings.customer_emails.length > 0) {
    const emails = settings.customer_emails.join(' OR ');
    searchParts.push(`(from:(${emails}) OR to:(${emails}))`);
  }
  
  // Add keyword filters (in subject or body)
  if (settings.customer_keywords.length > 0) {
    const keywords = settings.customer_keywords.join(' OR ');
    searchParts.push(`(${keywords})`);
  }
  
  // Add date filter (last 40 days)
  const fortyDaysAgo = new Date();
  fortyDaysAgo.setDate(fortyDaysAgo.getDate() - 40);
  const dateStr = fortyDaysAgo.toISOString().split('T')[0].replace(/-/g, '/');
  searchParts.push(`after:${dateStr}`);
  
  const searchQuery = searchParts.join(' ');
  const encodedQuery = encodeURIComponent(searchQuery);
  
  return `https://mail.google.com/mail/u/0/#search/${encodedQuery}`;
}
